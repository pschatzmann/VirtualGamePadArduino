172 void handleClient() {
173 static std::vector<uint8_t> buffer;
174 auto client = _server->available();
175 if (client) {
176 while (client.connected()) {
177 while (client.available()) {
178 uint8_t byte = client.read();
179 buffer.push_back(byte);
180 }
181 processBuffer(buffer);
182 delay(10);
183 }
184 client.stop();
185 }
186 }

191 private:
192 NetworkServer* _server = nullptr;
193 GamepadStateCallback _callback = nullptr;
194 GamepadState state;
195
196 // Helper: process as many complete packets as possible from the buffer
197 void processBuffer(std::vector<uint8_t>& buffer) {
198 while (!buffer.empty()) {
199 vgp_data_exchange_gamepad_reading reading;
200 memset(&reading, 0, sizeof(reading));
201 size_t used = vgp_data_exchange_gamepad_reading_unmarshal(
202 &reading, buffer.data(), buffer.size());
203 if (used == 0) {
204 if (errno == EWOULDBLOCK) {
205 // Incomplete data, wait for more
206 break;
207 } else if (errno == EFBIG || errno == EILSEQ) {
208 // Data too large or schema mismatch, drop first byte and try again
209 buffer.erase(buffer.begin());
210 continue;
211 } else {
212 // Unknown error, clear buffer
213 buffer.clear();
214 break;
215 }
216 } else {
217 // Success
218 state = GamepadState(reading);
219 if (_callback) _callback(state);
220 buffer.erase(buffer.begin(), buffer.begin() + used);
221 }
222 }
223 }
